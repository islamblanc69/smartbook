// دالة فتح القائمة المنسدلة مع إغلاق البقية تلقائياً function toggleAccordion(element) { const currentContent = element.nextElementSibling; const currentIcon = element.querySelector('span:last-child'); // 1. البحث عن أي قائمة مفتوحة حالياً وإغلاقها document.querySelectorAll('.accordion-content').forEach(content => { if (content !== currentContent) { content.style.maxHeight = '0px'; content.style.paddingTop = '0px'; content.style.paddingBottom = '0px'; // تحويل شارة الـ المفتوحة إلى زائد مجدداً const header = content.previousElementSibling; header.querySelector('span:last-child').innerText = '➕'; } }); // 2. التحكم في القائمة الحالية التي ضغط عليها المستخدم (فتح أو إغلاق) if (currentContent.style.maxHeight && currentContent.style.maxHeight !== '0px') { currentContent.style.maxHeight = '0px'; currentContent.style.paddingTop = '0px'; currentContent.style.paddingBottom = '0px'; currentIcon.innerText = '➕'; } else { currentContent.style.maxHeight = currentContent.scrollHeight + 40 + "px"; currentContent.style.paddingTop = '10px'; currentContent.style.paddingBottom = '15px'; currentIcon.innerText = '➖'; } }
